		\glidera{0.035}%
	}{%
		\ifthenelse{\equal{\intcalcMod{\value{#1}}{5}}{2}}{
			\gliderb{0.035}%
		}{%
			\ifthenelse{\equal{\intcalcMod{\value{#1}}{5}}{3}}{
				\gliderc{0.035}%
			}{%
				\ifthenelse{\equal{\intcalcMod{\value{#1}}{5}}{4}}{
					\gliderd{0.035}%
				}{%
					\glidere{0.035}%
				}%
			}%
		}%
	}%
